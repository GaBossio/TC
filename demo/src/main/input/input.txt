#include <stdio.h>
#include "math.h"

// ========== CASOS VÁLIDOS ==========

// Función principal
int main()
{
    // Declaraciones de variables
    int x;
    float y;
    double z;
    char c;
    string s;
    bool flag;
    
    // Asignaciones básicas
    x = 10;
    y = 3.14;
    z = 2.718281;
    c = 'A';
    s = "Hola mundo";
    flag = true;
    
    // Expresiones aritméticas
    x = x + 5;
    y = x * y - 2.5;
    z = (x + y) / 2.0;
    
    // Operadores de comparación
    if (x > 10)
    {
        y = y + 1.0;
    }
    else
    {
        y = y - 1.0;
    }
    
    // Operadores lógicos
    if (x >= 5 && y <= 20.0)
    {
        flag = false;
    }
    
    if (flag || x != 0)
    {
        z = z * 2.0;
    }
    
    // Bucle while
    while (x > 0)
    {
        x = x - 1;
        if (x == 5)
        {
            break;
        }
        if (x % 2 == 0)
        {
            continue;
        }
    }
    
    // Bucle for
    for (int i = 0; i < 10; i++)
    {
        y = 0;
        y = y + i;
    }
    
    // Asignaciones compuestas
    x += 5;
    y -= 2.5;
    z *= 1.5;
    x /= 2;
    x %= 3;
    
    // Operadores unarios
    x++;
    --y;
    ++z;
    x--;
    
    // Expresiones complejas
    int resultado = (x + y) * (z - 1.0) / (x != 0 ? x : 1);
    
    return 0;
}

// Función void
void imprimir(string mensaje)
{
    // Variables locales
    char prefijo;
    bool mostrar;
    
    prefijo = '[';
    mostrar = true;
    int valor = 0;
    
    if (mostrar)
    {
        // Usar parámetros
        valor = valor + 1;
    }
}
